    pure function char_array_to_int(str_array) result(i)
        character(len=1),dimension(:),intent(in) :: str_array !! example ['1','3'] --> 13
        integer :: i
        character(len=:),allocatable :: str
        integer :: k
        str = ''
        do k = 1, size(str_array)
            str = str//str_array(k)
        end do
        i = char_to_int(str)
    end function char_array_to_int
